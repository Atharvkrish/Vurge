6. Database Changes (Run Migrations)
Add the following new models to prisma/schema.prisma:

prisma
model University {
  id       String   @id @default(cuid())
  name     String   @unique
  domain   String   @unique   // e.g., "tcd.ie"
  createdAt DateTime @default(now())
}

model UniversityPlacement {
  id          String   @id @default(cuid())
  universityId String
  university   University @relation(fields: [universityId], references: [id])
  employerId   String
  employer     EmployerProfile @relation(fields: [employerId], references: [id])
  jobId        String
  job          Job      @relation(fields: [jobId], references: [id])
  active       Boolean  @default(true)
  createdAt    DateTime @default(now())
}

model DirectApplication {
  id          String   @id @default(cuid())
  studentId   String
  student     StudentProfile @relation(fields: [studentId], references: [id])
  placementId String
  placement   UniversityPlacement @relation(fields: [placementId], references: [id])
  status      ApplicationStatus @default(PENDING)
  appliedAt   DateTime @default(now())
}
Then run:
